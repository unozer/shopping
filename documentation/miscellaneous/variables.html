authInterceptor
Type : HttpInterceptorFn
Default value : (req, next) => { const authService = inject(AuthService); const authReq = req.clone({ setHeaders: { Authorization: 'myToken' }, }); return next(authReq).pipe( catchError((error: HttpErrorResponse) => { if (error.status === HttpStatusCode.Unauthorized) { authService.logout(); return EMPTY; } else { return throwError(() => error); } }) ); }
